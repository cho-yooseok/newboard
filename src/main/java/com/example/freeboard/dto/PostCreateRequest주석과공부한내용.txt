주석
package com.example.freeboard.dto;

import jakarta.validation.constraints.NotBlank;
import jakarta.validation.constraints.Size;
import lombok.Getter;
import lombok.Setter;

// 게시글 생성 요청을 처리하기 위한 DTO (Data Transfer Object)
// 클라이언트에서 전달한 "제목"과 "내용" 데이터를 담는 역할
@Getter // lombok: 모든 필드의 getter 메서드를 자동 생성
@Setter // lombok: 모든 필드의 setter 메서드를 자동 생성
public class PostCreateRequest {

    // 제목(title) 필드
    @NotBlank(message = "제목은 필수입니다.")
    // -> null, 빈 문자열(""), 공백("  ") 모두 허용하지 않음
    // -> 유효성 검사 실패 시 "제목은 필수입니다." 라는 메세지가 반환됨

    @Size(max = 255, message = "제목은 최대 255자까지 가능합니다.")
    // -> 제목의 글자 수가 255자를 초과하면 에러 발생
    private String title;

    // 내용(content) 필드
    @NotBlank(message = "내용은 필수입니다.")
    // -> 반드시 내용이 있어야 함 (빈 값 불가)

    @Size(max = 65535, message = "내용은 최대 65535자 까지 가능합니다.")
    // -> 내용은 TEXT 타입 (DB 기준)에 맞춰 최대 65,535 자로 제한
    private String content;
}

해설
1. DTO의 역할
이 클래스는 사용자가 게시글을 작성할 때 HTTP 요청(예 : JSON)으로 전달되는 데이터를 받는 그릇 역할을 합니다.
